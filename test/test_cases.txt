# basic arithmetic
2 + 3 = 5
5 * 4 = 20
10 - 3 = 7
15 / 3 = 5

# operator precedence
2 + 3 * 4 = 14
(2 + 3) * 4 = 20
2 * 3 + 4 = 10
2 + 3 + 4 = 9

# power operator
2 ** 3 = 8
3 ** 2 = 9
2 ** 0 = 1
0 ** 2 = 0

# negative numbers
-5 = -5
-5 + 3 = -2
3 * -2 = -6
-2 ** 3 = -8

# parentheses
(1 + 2) * 3 = 9
(2 * 3) + 4 = 10
((1 + 2) * 3) + 4 = 13
